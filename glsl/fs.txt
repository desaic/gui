#version 330
in vec3 pos;
in vec3 normal;
out vec4 outcolor;
uniform vec3 lightpos;
void main()
{
    vec3 L = vec3(0.0, 1.0, 0.0) - pos;
	L = normalize(L);
	vec3 N = normalize(normal);
	float diffuse = 0.5 * max(0, dot(L, N));
	vec3 V = -normalize(pos);
	vec3 H = L + V;
	H = normalize(H);
	float alpha=4;
	float specular = max(0,dot(N,H));
	specular = 0.5* pow(specular, alpha);
    vec4 c = (diffuse + specular )* vec4(1,1,1, 1.0);
	c.w=1.0;
    outcolor = c;
};
